Roblox Auto Play Script

Draggable UI with Auto Play & Float Features

📜 Script Code 
local UIS = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local Players = game:GetService("Players")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")
local rootPart = character:WaitForChild("HumanoidRootPart")

-- UI Root
local screenGui = Instance.new("ScreenGui", player.PlayerGui)
screenGui.Name = "AutoPlayHub"
screenGui.ResetOnSpawn = false

-- Main Frame (Right Side)
local mainFrame = Instance.new("Frame", screenGui)
mainFrame.Size = UDim2.new(0, 240, 0, 200)
mainFrame.Position = UDim2.new(1, -260, 0.5, -100)
mainFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
mainFrame.Active = true
mainFrame.Draggable = true

local uiCorner = Instance.new("UICorner", mainFrame)
uiCorner.CornerRadius = UDim.new(0, 10)

-- Animated Gradient Outline
local outline = Instance.new("Frame", mainFrame)
outline.Size = UDim2.new(1, 6, 1, 6)
outline.Position = UDim2.new(0, -3, 0, -3)
outline.ZIndex = 0
local outCorner = Instance.new("UICorner", outline)
outCorner.CornerRadius = UDim.new(0, 10)

local grad = Instance.new("UIGradient", outline)
grad.Color = ColorSequence.new(Color3.fromRGB(0, 170, 255), Color3.fromRGB(0, 102, 255))
task.spawn(function()
    while task.wait(0.03) do 
        grad.Rotation += 2
    end
end)

-- Title "Auto Play" with Gradient
local title = Instance.new("TextLabel", mainFrame)
title.Text = "Auto Play"
title.Size = UDim2.new(1, -20, 0, 35)
title.Position = UDim2.new(0, 10, 0, 10)
title.BackgroundTransparency = 1
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.TextSize = 20
title.Font = Enum.Font.GothamBold
title.TextXAlignment = Enum.TextXAlignment.Center

local titleGrad = Instance.new("UIGradient", title)
titleGrad.Color = ColorSequence.new(Color3.fromRGB(0, 170, 255), Color3.fromRGB(0, 102, 255))

-- Auto Play Button
local autoPlayBtn = Instance.new("TextButton", mainFrame)
autoPlayBtn.Name = "AutoPlayButton"
autoPlayBtn.Size = UDim2.new(0, 200, 0, 40)
autoPlayBtn.Position = UDim2.new(0.5, -100, 0, 55)
autoPlayBtn.Text = "Auto Play"
autoPlayBtn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
autoPlayBtn.TextColor3 = Color3.new(1, 1, 1)
autoPlayBtn.Font = Enum.Font.GothamBold
autoPlayBtn.TextSize = 16
Instance.new("UICorner", autoPlayBtn).CornerRadius = UDim.new(0, 8)

-- Button Outline (Hidden Initially)
local btnOutline = Instance.new("Frame", autoPlayBtn)
btnOutline.Size = UDim2.new(1, 6, 1, 6)
btnOutline.Position = UDim2.new(0, -3, 0, -3)
btnOutline.ZIndex = -1
btnOutline.BackgroundTransparency = 1
local btnOutCorner = Instance.new("UICorner", btnOutline)
btnOutCorner.CornerRadius = UDim.new(0, 8)

local btnGrad = Instance.new("UIGradient", btnOutline)
btnGrad.Color = ColorSequence.new(Color3.fromRGB(0, 170, 255), Color3.fromRGB(0, 102, 255))

-- Float Label
local floatLabel = Instance.new("TextLabel", mainFrame)
floatLabel.Text = "Float"
floatLabel.Size = UDim2.new(0, 80, 0, 30)
floatLabel.Position = UDim2.new(0, 20, 0, 110)
floatLabel.BackgroundTransparency = 1
floatLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
floatLabel.TextSize = 14
floatLabel.Font = Enum.Font.GothamBold
floatLabel.TextXAlignment = Enum.TextXAlignment.Left

local floatLabelGrad = Instance.new("UIGradient", floatLabel)
floatLabelGrad.Color = ColorSequence.new(Color3.fromRGB(0, 170, 255), Color3.fromRGB(0, 102, 255))

-- Toggle Switch for Float
local toggleBack = Instance.new("Frame", mainFrame)
toggleBack.Size = UDim2.new(0, 50, 0, 25)
toggleBack.Position = UDim2.new(0, 110, 0, 113)
toggleBack.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
Instance.new("UICorner", toggleBack).CornerRadius = UDim.new(1, 0)

local toggleCircle = Instance.new("Frame", toggleBack)
toggleCircle.Size = UDim2.new(0, 21, 0, 21)
toggleCircle.Position = UDim2.new(0, 2, 0, 2)
toggleCircle.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
Instance.new("UICorner", toggleCircle).CornerRadius = UDim.new(1, 0)

-- Toggle Button (Invisible but clickable)
local toggleBtn = Instance.new("TextButton", toggleBack)
toggleBtn.Size = UDim2.new(1, 0, 1, 0)
toggleBtn.BackgroundTransparency = 1
toggleBtn.Text = ""

-- Status Text
local statusText = Instance.new("TextLabel", mainFrame)
statusText.Text = "Status: Idle"
statusText.Size = UDim2.new(1, -20, 0, 20)
statusText.Position = UDim2.new(0, 10, 1, -30)
statusText.BackgroundTransparency = 1
statusText.TextColor3 = Color3.fromRGB(150, 150, 150)
statusText.TextSize = 11
statusText.Font = Enum.Font.Gotham

--- LOGIC ---
local autoPlayEnabled = false
local floatEnabled = false
local bPos = nil
local savedAnimations = {}
local autoPlayConnection = nil

-- Animation Removal Function
local function removeAnimations()
    savedAnimations = {}
    for _, track in pairs(humanoid:GetPlayingAnimationTracks()) do
        if track.Name == "WalkAnim" or track.Name == "RunAnim" or track.Name == "JumpAnim" then
            table.insert(savedAnimations, track)
            track:Stop()
        end
    end
    
    -- Disable default animations
    local animate = character:FindFirstChild("Animate")
    if animate then
        animate.Disabled = true
    end
end

-- Restore Animations Function
local function restoreAnimations()
    local animate = character:FindFirstChild("Animate")
    if animate then
        animate.Disabled = false
    end
end

-- Auto Play Button Click
autoPlayBtn.MouseButton1Click:Connect(function()
    autoPlayEnabled = not autoPlayEnabled
    
    if autoPlayEnabled then
        autoPlayBtn.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
        btnOutline.BackgroundTransparency = 0
        statusText.Text = "Status: Auto Play Active"
        statusText.TextColor3 = Color3.fromRGB(0, 255, 100)
        
        -- Animate button outline
        task.spawn(function()
            while autoPlayEnabled and task.wait(0.03) do
                btnGrad.Rotation += 2
            end
        end)
        
        -- Remove animations
        removeAnimations()
        
        -- Auto Play Logic
        autoPlayConnection = RunService.Heartbeat:Connect(function()
            if not autoPlayEnabled then return end
            
            local currentPos = rootPart.Position
            local x = currentPos.X
            
            -- Determine direction based on X coordinate
            -- If X < 0, move left; if X > 0, move right
            if x < 0 then
                -- Move Left
                humanoid:Move(Vector3.new(-1, 0, 0), false)
            else
                -- Move Right
                humanoid:Move(Vector3.new(1, 0, 0), false)
            end
        end)
    else
        autoPlayBtn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        btnOutline.BackgroundTransparency = 1
        statusText.Text = "Status: Idle"
        statusText.TextColor3 = Color3.fromRGB(150, 150, 150)
        
        -- Restore animations
        restoreAnimations()
        
        -- Disconnect auto play
        if autoPlayConnection then
            autoPlayConnection:Disconnect()
            autoPlayConnection = nil
        end
    end
end)

-- Float Toggle Click
toggleBtn.MouseButton1Click:Connect(function()
    floatEnabled = not floatEnabled
    
    if floatEnabled then
        toggleBack.BackgroundColor3 = Color3.fromRGB(0, 170, 255)
        toggleCircle:TweenPosition(UDim2.new(1, -23, 0, 2), "Out", "Quad", 0.2, true)
        statusText.Text = "Status: Floating"
        statusText.TextColor3 = Color3.fromRGB(0, 200, 255)
        
        -- Create BodyPosition for floating
        bPos = Instance.new("BodyPosition")
        bPos.MaxForce = Vector3.new(0, math.huge, 0)
        bPos.P = 10000
        bPos.D = 600
        bPos.Position = rootPart.Position + Vector3.new(0, 10, 0)
        bPos.Parent = rootPart
    else
        toggleBack.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        toggleCircle:TweenPosition(UDim2.new(0, 2, 0, 2), "Out", "Quad", 0.2, true)
        
        if not autoPlayEnabled then
            statusText.Text = "Status: Idle"
            statusText.TextColor3 = Color3.fromRGB(150, 150, 150)
        end
        
        -- Remove BodyPosition
        if bPos then 
            bPos:Destroy() 
            bPos = nil 
        end
    end
end)

-- Float Movement Update
RunService.Heartbeat:Connect(function()
    if floatEnabled and bPos and rootPart then
        local ray = Ray.new(rootPart.Position + Vector3.new(0, 2, 0), Vector3.new(0, -500, 0))
        local hit, pos = workspace:FindPartOnRay(ray, character)
        
        if hit then
            bPos.Position = Vector3.new(rootPart.Position.X, pos.Y + 10, rootPart.Position.Z)
        end
    end
end)

print("Auto Play script loaded successfully!")
