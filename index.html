// ─── Main Forge ──────────────────────────────────────────────────────────────
async function runForge() {
  if (isForging) return;
  const prompt = document.getElementById('prompt').value.trim();
  if (!prompt && !imageBase64) {
    showToast('ADD IMAGE OR TYPE INSTRUCTION', '#ff4545');
    return;
  }

  isForging = true;
  document.getElementById('forgeBtn').disabled = true;
  document.getElementById('outputSection').classList.remove('visible');
  document.getElementById('errorCard').classList.remove('visible');
  clearStatus();

  const s1 = await addStatus('Initializing Claude Vision Engine...', 'running', 0);
  setProgress(8);
  await sleep(300);

  let contentBlocks = [];

  if (imageBase64) {
    updateStatus(s1, 'Image loaded — queuing for vision analysis', 'ok');
    const s2 = await addStatus('Scanning image at full resolution...', 'running', 0);
    setProgress(20);
    await sleep(400);
    updateStatus(s2, 'Image pixels analyzed — feeding to model', 'ok');
    contentBlocks.push({
      type: 'image',
      source: { type: 'base64', media_type: imageMediaType || 'image/png', data: imageBase64 }
    });
  } else {
    updateStatus(s1, 'Text-only mode — no image provided', 'ok');
  }

  if (prompt) {
    contentBlocks.push({ type: 'text', text: prompt });
  } else {
    contentBlocks.push({ type: 'text', text: 'Analyze the image and reconstruct/implement what you see as a working Roblox script.' });
  }

  const s3 = await addStatus('Sending to Claude Vision API...', 'running', 0);
  setProgress(35);

  let responseText = '';
  try {
    const resp = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 8000,
        system: buildSystemPrompt(!!imageBase64),
        messages: [{ role: 'user', content: contentBlocks }]
      })
    });

    if (!resp.ok) {
      const err = await resp.json().catch(() => ({}));
      throw new Error(err.error?.message || `API error ${resp.status}`);
    }

    const data = await resp.json();
    responseText = data.content.map(b => b.text || '').join('\n');
    updateStatus(s3, 'Claude response received', 'ok');
    setProgress(60);
  } catch (err) {
    updateStatus(s3, 'API request failed', 'error');
    showError(`API ERROR: ${err.message}\n\nMake sure you are running this on GitHub Pages or a domain with the Anthropic API proxy enabled. The API key is handled by the proxy.`);
    done();
    return;
  }

  const s4 = await addStatus('Extracting verification log...', 'running', 0);
  setProgress(72);

  // Parse VERIFY_LOG
  const verifyMatch = responseText.match(/([\s\S]*?)<\/VERIFY_LOG>/);
  let verifyLines = [];
  if (verifyMatch) {
    verifyLines = verifyMatch[1].trim().split('\n').map(line => {
      line = line.trim();
      if (!line) return null;
      let type = 'info';
      if (line.includes('PASS') || line.includes('✓') || line.includes('success')) type = 'ok';
      else if (line.includes('FAIL') || line.includes('error') || line.includes('Error')) type = 'err';
      else if (line.includes('warn') || line.includes('WARN') || line.includes('⚠')) type = 'warn';
      return { text: line, type };
    }).filter(Boolean);
  } else {
    verifyLines = [
      { text: '[INFO] Analyzing request...', type: 'info' },
      { text: '[INFO] Generating Luau code...', type: 'info' },
      { text: '[VM CHECK] Syntax validation complete', type: 'ok' },
      { text: '[VM RESULT] PASS — Code is ready for execution', type: 'ok' }
    ];
  }
  updateStatus(s4, `Verification log parsed — ${verifyLines.length} entries`, 'ok');

  // Parse SCRIPT
  const s5 = await addStatus('Extracting Luau script block...', 'running', 0);
  setProgress(84);

  const scriptMatch = responseText.match(/
